Virtual Dom 


example:

import { createRoot } from "react-dom/client";

const rootElement = document.getElementById("root");
const root = createRoot(rootElement);

const update = () => {
  const element = (
    <>
      <h3>React:</h3>
      <form>
        <input type="text" />
      </form>
      <span>Time: {new Date().toLocaleTimeString()}</span>
    </>
  );
  // console.log(element);
  root.render(element);
};

setInterval(update, 1000);
